rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    // Règles pour les utilisateurs
    match /users/{userId} {
      // Permettre la lecture et l'écriture pour tous (temporaire pour le développement)
      allow read, write: if true;
    }
    
    // Règles pour les commandes
    match /orders/{orderId} {
      allow read, write: if true;
    }
    
    // Règles pour les cartes de visite
    match /businessCards/{cardId} {
      allow read, write: if true;
    }
    
    // Règles pour les produits
    match /products/{productId} {
      allow read, write: if true;
    }

    // Règles pour les documents
    match /documents/{documentId} {
      allow read, write: if true;
    }
    
    // Règles pour les scans QR (NOUVELLES - À AJOUTER)
    match /qrScans/{scanId} {
      allow read, write: if true;
    }
    
    // Règles pour les téléchargements (NOUVELLES - À AJOUTER)
    match /downloads/{downloadId} {
      allow read, write: if true;
    }
  }
}
